```c
int main()
{
    int n;
    scanf("%d",&n);
    int array[n];
    for(int i=0; i<n; i++)
    {
        scanf("%d",&array[i]);
    }

    int smallest=array[0];
    for(int i=1; i<n ; i++)
    {
        if(array[i]<smallest)
        {
            smallest=array[i];
        }
    }
    printf("%d",smallest);
    return 0;
}
```
